import express from "express"; import bodyParser from "body-parser"; import cors from "cors"; import fetch from "node-fetch"; const app = express(); const PORT = process.env.PORT || 3000; app.use(cors()); app.use(bodyParser.json()); // Health check app.get("/", (req, res) => { res.send("✅ fb-proxy is running!"); }); // Main API app.post("/post-now", async (req, res) => { try { const { topic, count, provider } = req.body; if (!topic) { return res.status(400).json({ error: "Missing topic" }); } // --- เรียก Gemini API --- if (provider === "gemini") { const response = await fetch( "https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=" + process.env.GEMINI_API_KEY, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: `เขียนโพสต์ Facebook เรื่อง: ${topic}` }] }] }) } ); const data = await response.json(); return res.json({ result: data }); } // --- เผื่อรองรับ OpenAI (ยังไม่ set key ก็ไม่เป็นไร) --- if (provider === "openai") { if (!process.env.OPENAI_API_KEY) { return res.json({ error: "OPENAI_API_KEY not set" }); } // ... code เรียก OpenAI ต่อ } res.json({ error: "Unknown provider" }); } catch (err) { res.status(500).json({ error: err.message }); } }); app.listen(PORT, () => { console.log(`🚀 Server running on port ${PORT}`); });
